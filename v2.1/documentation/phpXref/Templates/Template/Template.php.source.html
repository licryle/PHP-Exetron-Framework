 177      /**
 178       * Generates a printable version of object for final print out.
 179       * It replaces each tag by it's Template Generated value.
 180       * So it generate final document by hierarchy.
 181       *
 182       * @param $cached boolean that should always be false. It enables caching
 183       * of generated values for future call. Due to internal generation by hierarchy,
 184       * the function always call herself with $cached as true
 185       *
 186       * @return printable version of document
 187       *
 188       */
 189      public function Generate( $cached = false )
 190      {
 191          if ( $cached && ! empty ( $this->lastGenerated ) )
 192          {
 193              return $this->lastGenerated;
 194          }
 195      
 196          $generated = $this->skeleton;
 197          $lastGenerated = '';
 198  
 199          while ( $generated != $lastGenerated )
 200          {
 201              $lastGenerated = $generated;
 202              
 203              foreach ( $this->tags as $tag => $value )
 204              // replace tags by value, generated by subtemplates...
 205              {
 206                  // generation by hierarchy
 207                  $val = $value->Generate( true );
 208                  $generated = str_replace ( $tag, $val, $generated );
 209              }
 210          }
 211  
 212          $this->lastGenerated = $generated;
 213          
 214          return $generated;
 215      } //----- End of Generate
