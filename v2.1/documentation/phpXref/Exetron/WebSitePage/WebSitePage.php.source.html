 243  	protected function initializeDatabasesConnections( & $DBVars, $confDBIndex )
 244      {        
 245          $this->confDatabaseConnection = NULL;
 246      
 247          // set up all 'always' needed connections
 248          foreach ( $DBVars as $index => $db )
 249          {
 250              if ( $db['alwaysUse'] )
 251              {
 252                  // set up
 253                  $className = $db[ 'type' ].'Connection';
 254                  $newDB = new $className ( ) or ($errors = new Errors());
 255                  
 256                  $newDB->SetServer( $db['host'] );
 257                  $newDB->SetUsername( $db['login'] );
 258                  $newDB->SetPassword( $db['password'] );
 259                  
 260                  $newDB->Open( BDDConnection::CONNECTION_PERSISTENT );
 261                  
 262                  $errors = $newDB->SetDatabase( $db['database'] );
 263                  
 264                  if ( $errors == NULL )
 265                  {
 266                      $this->databasesConnections[] = & $newDB;
 267                      
 268                      if ( $index == $confDBIndex )
 269                      {

 272                  }
 273                  else
 274                  {

 276                  }
 277
