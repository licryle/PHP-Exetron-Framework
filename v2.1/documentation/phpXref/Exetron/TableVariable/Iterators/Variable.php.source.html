  44       * @param $siteTable a BDDTableSite object (where BDD should be

  47       *
  48       * @return - NULL if object has been validated
  49       * @return - an Errors object in case of error(s) :
  50       *
  51       * @return VariableError::VARIABLE_NAME_EMPTY if property 
  52       * TableVariable::TABLE_COLUMN_NAME is empty
  53       *
  54       * @return BDDError::TABLE_CLASS_INCORRECT if $groupTable is not a 
  55       * valid instance

  57       * @return VariableError::VARIABLE_SCOPE_INCORRECT if property

  63       * @return VariableError::VARIABLE_IDSITE_INEXISTANT if property 
  64       * TableVariable::TABLE_COLUMN_IDSITE refers to a non existant site
  65       */
  66      public function Validate ( $siteTable )
  67      {
  68          $errors = new Errors ();
  69      
  70          // variable name
  71              if ( empty( $this->row[ TableVariable::TABLE_COLUMN_NAME ] ) )
  72              {
  73                  $errors->Add ( new VariableError ( VariableError::VARIABLE_NAME_EMPTY, 'Please fill in variable name.') );
  74              }
  75      
  76          // scope
  77              if ( $this->row[ TableVariable::TABLE_COLUMN_SCOPE ] != TableVariable::TABLE_COLUMN_SCOPE_SERVER && $this->row[ TableVariable::TABLE_COLUMN_SCOPE ] != TableVariable::TABLE_COLUMN_SCOPE_SITE  )
  78              {
  79                  $errors->Add ( new VariableError ( VariableError::VARIABLE_SCOPE_INCORRECT, 'Variable scope is incorrect.') );
  80              }
  81          
  82          // access
  83              if ( $this->row[ TableVariable::TABLE_COLUMN_ACCESS ] != TableVariable::TABLE_COLUMN_ACCESS_ROOT && $this->row[ TableVariable::TABLE_COLUMN_ACCESS ] != TableVariable::TABLE_COLUMN_ACCESS_ADMIN )
  84              {
  85                  $errors->Add ( new VariableError ( VariableError::VARIABLE_ACCESS_INCORRECT, 'Variable access are not correctly defined.') );
  86              }
  87          
  88          // referent IdSite
  89              if ( ! @in_array( 'TableSiteInterface', class_implements ( $siteTable ) ) )
  90              {
  91                  $errors->Add( new BDDError ( BDDError::TABLE_CLASS_INCORRECT , 'Parameter is not a good instance of BDDTableSite.' ) );
  92              } 
  93              else
  94              {    
  95                  if ( ! $siteTable->IdSiteExists( $this->row[ TableVariable::TABLE_COLUMN_IDSITE ]  ) )
  96                  {
  97                      $errors->Add ( new VariableError ( VariableError::VARIABLE_IDSITE_INEXISTANT, 'Variable does not refer to any site.') );
  98                  }
  99              }
 100          
 101          // result
 102          if ( $errors->GetCount() == 0 )
 103          {
 104              $this->isValid = true;
 105              return NULL;        
 106          }
 107          
 108          
 109          $this->isValid = false;
 110          return $errors;
 111      } //---- End of Validate
