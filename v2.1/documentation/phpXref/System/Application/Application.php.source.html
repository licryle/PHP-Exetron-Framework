  31  /*!
  32   * Represents an application. All your PHP applications may inherits from
  33   * this class. It has to be instancied by a first call to GetInstance().
  34   * All next calls will only return the reference to the Application pre-
  35   * viously instancied.
  36   *
  37   * When application is started (a call to Start()), the Application object
  38   * automatically calls the call back set up with OnApplicationStart().
  39   *
  40   * A call back is also called when application ends. Application uses the
  41   * PHP's API register_shudown_function. You MUST NOT use this function by
  42   * yourself as it would by pass Application class's callback.
  43   *
  44   * Application can keep the configuration of your application by using 
  45   * SetConfiguration() and GetConfiguration().
  46   *
  47   * As a general abstraction of an application, Application does not
  48   * include $_GET, $_POST or $_FILES variables. You may look at child named

 103          if ( $configuration == NULL )

 115      /**
 116       * Start the application.
 117       * Call the call back function set up by OnApplicationStart.
 118       *
 119       * @return null if application successfully started
 120       * @return an object of type Errors in case of error(s).
 121       *
 122       */
 123      public function Start ( )
 124      {
 125          if ( $this->started )
 126          {
 127              $errors = new Errors ();
 128              $errors->Add ( new ApplicationError ( ApplicationError::ALREADY_STARTED, 'Application has been already started.' ) );
 129              
 130              return $errors;
 131          }
 132      
 133          // get launchTime in microseconds
 134          $this->systemVars[ self::SYSTEM_START_TIME ] = microtime ( true );
 135          
 136          // set up call back function for the end of the application
 137          register_shutdown_function ( $this->onApplicationEnd, $this->systemVars );
 138  
 139          return $this->launchCallBack ( 'onApplicationStart' );
 140      } //----- End of Start

 142      /**
 143       * Set up call_back function for Application Start.
 144       * The $function will be called when Start() will.
 145       * 
 146       * @param $function the name of the function to be called
 147       * @param $params must be an array of parameters for the function
 148       * 
 149       * For calling method of a class, use array (& $obj, 'method_name')
 150       * 
 151       * @return an object of Errors' type in case of Error(s)
 152       * @return null instead
 153       * 
 154       * 
 155       * Errors may be composed by :
 156       * - ApplicationError::FUNCTION_NOT_CALLABLE;
 157       * - ApplicationError::FUNCTION_PARAM_NOT_ARRAY;
 158       * - ApplicationError::CALLBACK_NOT_EXISTS;
 159       * 
 160       */
 161      public function OnApplicationStart ( $function, $params )
 162      {
 163          return $this->setCallBack ( 'onApplicationStart', $function, $params ) ;
 164      } //----- End of OnApplicationStart

 166      /**
 167       * Set up call_back function for Application End.
 168       * The $function will be called when Script will be stopped
 169       * 
 170       * @param $function the name of the function to be called
 171       * @param $function may accept 1 argument. This argument will be an array of system variables
 172       * Please look at the definition of systemVars
 173       * 
 174       * For calling method of a class, use array (& $obj, 'method_name')
 175       * 
 176       * @return an object of Errors' type in case of Error(s)
 177       * @return null instead
 178       *
 179       * Errors may be composed by :
 180       * - ApplicationError::FUNCTION_NOT_CALLABLE;
 181       * - ApplicationError::FUNCTION_PARAM_NOT_ARRAY;
 182       * - ApplicationError::CALLBACK_NOT_EXISTS;
 183       *
 184       */
 185      public function OnApplicationEnd ( $function )
 186      {
 187          $falseArray = array();
 188              $errors = $this->setCallBack ( 'onApplicationEnd', $function, $falseArray ) ;
 189          unset( $falseArray );
 190          
 191          if ( $errors InstanceOf Errors )
 192          {
 193              return $errors;
 194          }

 196          return null;

 244      /**
 245       * Let us know if a $function and it's $params are good to form a 
 246       * call back function
 247       *
 248       * @param $function The function tobe verified
 249       * @param $params The parameters for the function as a call back
 250       * @param $errors The "return by reference"'s variable for errors detected.
 251       *
 252       * $errors MUST BE a valid Errors object.
 253       *
 254       * @return true if the function can be set as CallBack
 255       * @return false instead
 256       */
 257  	protected function isCorrectCallBack ( & $function, & $params, Errors & $errors )
 258      {    
 259          // function incorrect or doesn't have good scope
 260          if ( !is_callable( $function ) )
 261          {
 262              $errors->Add ( new ApplicationError ( ApplicationError::FUNCTION_NOT_CALLABLE, 'Incorrect function or scope.') );
 263          }
 264          
 265          // params not an array
 266          if ( !is_array( $params ) )
 267          {
 268              $errors->Add ( new ApplicationError ( ApplicationError::FUNCTION_PARAM_NOT_ARRAY, 'Params of the function must be an array.') );        
 269          }
 270          
 271          return ( $errors->GetCount () == 0 );
 272      } //------ End of isCorrectCallBack
 273      
 274      /**
 275       * Internally sets call back for callback named $callBackName/
 276       *
 277       * @param $callBackName the name of the call back to be set
 278       * @param $function the function to set as a callback
 279       * @param $params the parameters for the function to be set
 280       *
 281       * @return an object of Errors' type in case of Error(s)
 282       * @return null instead
 283       *
 284       * Errors may be composed by :
 285       * - ApplicationError::FUNCTION_NOT_CALLABLE;
 286       * - ApplicationError::FUNCTION_PARAM_NOT_ARRAY
 287       * - ApplicationError::CALLBACK_NOT_EXISTS
 288       *
 289       */
 290  	protected function setCallBack ( $callBackName, & $function, & $params )
 291      {            
 292          $errors = new Errors();
 293          
 294          if ( $this->isCorrectCallBack( $function , $params, $errors ) === false )
 295          // if we had some errors
 296          {
 297              return $errors;
 298          }
 299  
 300          if ( ! IsSet ( $this->{$callBackName} ) || ! IsSet ( $this->{$callBackName.'Params'} ) )
 301          {
 302              $errors->Add ( new ApplicationError ( ApplicationError::CALLBACK_NOT_EXISTS, 'You tried to set up a callback that doesn\'t exists.') );
 303              
 304              return $errors;
 305          }
 306          
 307          // association
 308          $this->{$callBackName} = $function;
 309          $this->{$callBackName.'Params'} = $params;
 310          
 311          unset ( $errors );
 312          
 313          return null;
 314      } //----- End of setCallBack
 315      
 316      /**
 317       * Launch CallBack function that names $callBackName.
 318       * 
 319       * @param $callBackName the name of the call back to be called
 320       *
 321       * @return an object of Errors' type in case of Error(s)
 322       * @return null instead
 323       *
 324       */
 325  	protected function launchCallBack ( $callBackName )
 326      {
 327          // launch start call_back function if set
 328          if ( @$this->{ $callBackName } !== -1 )
 329          {
 330              call_user_func_array ( $this->{ $callBackName }, $this->{ $callBackName.'Params' } );
 331              
 332              return null;
 333          }
 334          else
 335          {
 336              $errors = new Errors();
 337              $errors->Add ( new ApplicationError ( ApplicationError::CALLBACK_NOT_SET, 'You are trying to launch an unset callback function.') );
 338              
 339              return $errors;
 340          }
 341      } //----- End of launchCallBack
