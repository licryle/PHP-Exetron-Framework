  61      public function TableExists ( $table )
  62      {
  63          if ( $this->database == '' )
  64          // no database selected
  65          {
  66              $errs = new Errors();
  67  
  68              $errs->Add ( new BDDError ( BDDError::CONNECTION_NO_DB_SELECTED , 'No database selected.' ) );
  69              
  70              return $errs;            
  71          }
  72          else
  73          {
  74              if ( ( $tableList = & $this->Query( 'SHOW TABLES FROM '.$this->database ) instanceof Errors ) )
  75              // query failed
  76              {
  77                  return $tableList;
  78              }
  79              else
  80              // search for table
  81              {
  82                  foreach ( $tableList as $currentTable )
  83                  {
  84                      if ( $table == $currentTable->GetProperty( 'Tables_in_'.$this->database ) )
  85                      {
  86                          unset ( $tableList );
  87                          return True;
  88                      }
  89                  }
  90                  
  91                  $errs = new Errors();
  92      
  93                  $errs->Add ( new BDDError ( BDDError::CONNECTION_TABLE_INEXISTANT , 'This table does not exist.' ) );
  94                  
  95                  return $errs;

 230          if ( !$this->isConnected() ) 
 231          {
 232              if ( $isPersistent == BDDConnection::CONNECTION_PERSISTENT )
 233              {
 234                  $connectFunction = 'mysql_pconnect';

 236              else
 237              {
 238                  $connectFunction = 'mysql_connect';
 239              }
 240          
 241              if ( ($this->connection = @$connectFunction( $this->server , $this->username , $this->password )) )

 314       * tries to send a query to DB server

 317       * @param $query the query to be done

 325       *
 326       * @see http://dev.mysql.com/doc/refman/5.0/en/mysql-query.html
 327       *
 328       */
 329      public function Query( $query )
 330      {
 331          if ( $this->database == '' )
 332          // no database selected
 333          {
 334              $errs = new Errors();
 335  
 336              $errs->Add ( new BDDError ( BDDError::CONNECTION_NO_DB_SELECTED , 'Aucune base de données sélectionnée' ) );
 337              
 338              return $errs;            
 339          }
 340          else
 341          {
 342              if ( ( $result = @mysql_query( $query , $this->connection ) ) === false ) 
 343              // query failed
 344              {
 345                  $errs = new Errors();
 346      
 347                  $errs->Add ( new BDDError ( BDDError::CONNECTION_QUERY_FAILED , 'La requete a échoué ('.$query.') : '.mysql_error( $this->connection ) ) );
 348                  
 349                  return $errs;
 350              }
 351              else
 352              {
 353                  $this->nombreRequetes++;
 354              
 355                  // set in en BDDRecord
 356                  
 357                  $nouvellesRessources = new BDDRecordSet ( );
 358                  
 359                  while ( ($row = @mysql_fetch_assoc ( $result ) ) !== false )
 360                  {
 361                      $nouvellesRessources->Add ( new BDDRecord ( $row ) ) ;
 362                  }
 363                  
 364                  @mysql_free_result ( $result ); // free memory
 365  
 366                  return $nouvellesRessources ;
 367              }
 368          }
 369      } //----- End of Query
